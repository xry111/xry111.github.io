1 #include <bits/stdc++.h>
 2 using namespace std;
 3 
 4 int f[100*100000+1];
 5 int val[5000];
 6 
 7 int main()
 8 {
 9     int n, s;
10     while (scanf("%d%d", &n, &s)==2) {
11         memset(f, 0x3f, sizeof(f));
12         f[0] = 0;
13         for (int i = 0; i<n; i++) {
14             int v; scanf("%d", &v);
15             val[i] = v;
16             for (int j = 1; j<=s; j++)
17                 f[j*v] = min(f[j*v], j);
18         }
19         int q; scanf("%d", &q);
20         for (int i = 0; i<q; i++) {
21             int ans = 101;
22             int m; scanf("%d", &m);
23             for (int j = 0; j<n; j++)
24                 for (int k = 0; k<=(s+1)/2; k++) {
25                     if (m<val[j]*k) break;
26                     ans = min(ans, k+f[m-val[j]*k]);
27                 }
28             if (ans>s) puts("-1");
29             else printf("%d\n", ans);
30         }
31     }
32     return 0;
33 }
